CREATE DATABASE B
DEFAULT CHARSET UTF8
COLLATE UTF8_GENERAL_CI;

USE B;

CREATE TABLE vendedor(
idVendedor INT(11) UNSIGNET NOT NULL PRIMARY KEY UNIQUE AUTO_INCREMENT,
nome VARCHAR(45) NOT NULL
);

CREATE TABLE cliente(
idCliente  INT(11) UNSIGNET NOT NULL PRIMARY KEY UNIQUE AUTO_INCREMENT,
nome VARCHAR(45) NOT NULL
);

CREATE TABLE vendas(
fk_vendedor INT(11) UNSIGNED NOT NULL,
fk_cliente INT(11) UNSIGNED NOT NULL,
CONSTRAINT fk1
FOREIGN KEY (fk_vendedor)
REFERENCES vendedor(idVendedor),
CONSTRAINT fk2
FOREIGN KEY (fk_cliente)
REFERENCES cliente(idCliente)
);

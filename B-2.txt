CREATE DATABASE B
DEFAULT CHARSET UTF8
COLLATE UTF8_GENERAL_CI;

USE B;

CREATE TABLE pessoas(
id INT(11) UNSIGNED NOT NULL PRIMARY KEY UNIQUE AUTO_INCREMENT,
nome VARCHAR(45) NOT NULL,
tipo TINYINT(1) COMMENT '1=cliente, 2=vendedor'
);

CREATE TABLE venda(
fk_vendedor INT(11) UNSIGNED NOT NULL,  
fk_cliente INT(11) UNSIGNED NOT NULL,
CONSTRAINT FK1
FOREIGN KEY (fk_vendedor)
REFERENCES pessoas(id),
CONSTRAINT FK2
FOREIGN KEY (fk_cliente)
REFERENCES pessoas(id)
);
